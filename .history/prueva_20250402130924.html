// script.js document.addEventListener("DOMContentLoaded", () => { const dropdownTrigger = document.querySelector(".dropdown-trigger"); const dropdown = document.querySelector(".dropdown"); dropdownTrigger.addEventListener("click", () => { dropdown.classList.toggle("open"); }); // Close the dropdown if clicked outside document.addEventListener("click", (event) => { if (!dropdown.contains(event.target) && !dropdownTrigger.contains(event.target)) { dropdown.classList.remove("open"); } }); });
